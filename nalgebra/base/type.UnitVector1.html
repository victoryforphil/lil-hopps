Computes the spherical linear interpolation between two unit vectors.

Examples:

let v1 = Unit::new_normalize(Vector2::new(1.0, 2.0));
let v2 = Unit::new_normalize(Vector2::new(2.0, -3.0));

let v = v1.slerp(&v2, 1.0);

assert_eq!(v, v2);
